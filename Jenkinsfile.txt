    agent any
    stages {
        stage('Build Docker Image') {
            steps {
                bat 'docker build -t registration:v1 .'
            }
        }
        stage('Push to Docker Hub') {
            steps {
                bat 'docker tag registration:v1 umaponnolu/registration:v1'
                bat 'docker push umaponnolu/registration:v1'
            }
                       
        }

    }
}
